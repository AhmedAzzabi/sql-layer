use `coitest_1k`;

create table customers
(
	 customer_id int not null auto_increment,
	 primary key(customer_id),
	 key (customer_id),
     rand_id int,
	 customer_info varchar(100)
) engine=akibadb;

create table orders (
	 order_id int not null auto_increment,
	 primary key(order_id),
	 key (order_id),
	 customer_id int,
	 key (customer_id),
	 order_info varchar(200),
CONSTRAINT `__akiban_fk_0` FOREIGN KEY `__akiban_fk_0` (`customer_id`) REFERENCES `customers` (`customer_id`)
) engine=akibadb;

create table items (
	 item_id int not null auto_increment,
	 primary key(item_id),
	 key (item_id),
	 order_id int,
	 key (order_id),
	 item_info varchar(200),
CONSTRAINT `__akiban_fk_1` FOREIGN KEY `__akiban_fk_1` (`order_id`) REFERENCES `orders` (`order_id`)
) engine=akibadb;
